    <script>

        let appConfig = JSON.parse(localStorage.getItem('bsc_expert_v8')) || {

            widths: [25, 28, 30, 42, 45, 50, 55],

            freq: {

                natural: { rell: 3, full: 8 },

                sintetico: { rell: 5, full: 12 }

            }

        };

        let bikes = JSON.parse(localStorage.getItem('bikes_expert_v8')) || [];


        function calculateML(category, size, width) {

            let baseFactor = (size === '700c' || size === '29"') ? 2.1 : (size === '26"' ? 1.8 : 2.4);

            let ml = width * baseFactor;

            if (category === 'Montaña') ml *= 1.25;

            if (category === 'Gravel') ml *= 1.1;

            if (category === 'Ruta') ml *= 0.95;

            return Math.round(ml / 10) * 10;

        }


        function saveAppData() {

            localStorage.setItem('bsc_expert_v8', JSON.stringify(appConfig));

            localStorage.setItem('bikes_expert_v8', JSON.stringify(bikes));

            renderBikes();

        }


        function openBikeModal(index = -1) {

            document.getElementById('tireWidth').innerHTML = appConfig.widths.map(w => `<option value="${w}">${w} mm</option>`).join('');

            const today = new Date().toISOString().split('T')[0];

            document.getElementById('editIndex').value = index;

            if(index >= 0) {

                const b = bikes[index];

                document.getElementById('modalTitle').innerText = "EDITAR BICI";

                document.getElementById('bikeName').value = b.name;

                document.getElementById('bikeCategory').value = b.category;

                document.getElementById('tireSize').value = b.size;

                document.getElementById('tireWidth').value = b.width;

                document.getElementById('liquidType').value = b.liquid;

                document.getElementById('manualChangeDate').value = b.lastChange;

                document.getElementById('manualRefillDate').value = b.lastRefill;

            } else {

                document.getElementById('modalTitle').innerText = "NUEVA BICI";

                document.getElementById('bikeName').value = "";

                document.getElementById('manualChangeDate').value = today;

                document.getElementById('manualRefillDate').value = today;

            }

            document.getElementById('bikeModal').style.display = 'flex';

            updateMLSuggestions();

        }


        function saveBike() {

            const idx = parseInt(document.getElementById('editIndex').value);

            const data = {

                name: document.getElementById('bikeName').value || "BICI SIN NOMBRE",

                category: document.getElementById('bikeCategory').value,

                size: document.getElementById('tireSize').value,

                width: document.getElementById('tireWidth').value,

                liquid: document.getElementById('liquidType').value,

                lastChange: document.getElementById('manualChangeDate').value,

                lastRefill: document.getElementById('manualRefillDate').value

            };

            if(idx >= 0) bikes[idx] = data; else bikes.push(data);

            saveAppData(); closeModal('bikeModal'); showOk();

        }


        function quickUpdate(index, type) {

            const today = new Date().toISOString().split('T')[0];

            if (type === 'relleno') bikes[index].lastRefill = today;

            else { bikes[index].lastChange = today; bikes[index].lastRefill = today; }

            saveAppData(); showOk();

        }


        function updateMLSuggestions() {

            const cat = document.getElementById('bikeCategory').value;

            const size = document.getElementById('tireSize').value;

            const width = document.getElementById('tireWidth').value;

            const ml = calculateML(cat, size, width);

            document.getElementById('mlHint').innerText = `${ml} ML`;

        }


        function saveFrequencies() {

            appConfig.freq.natural.rell = parseInt(document.getElementById('freq_nat_rell').value);

            appConfig.freq.natural.full = parseInt(document.getElementById('freq_nat_full').value);

            appConfig.freq.sintetico.rell = parseInt(document.getElementById('freq_sin_rell').value);

            appConfig.freq.sintetico.full = parseInt(document.getElementById('freq_sin_full').value);

            saveAppData(); showOk();

        }


        function addWidth() {

            const val = parseInt(document.getElementById('newWidth').value);

            if(val && !appConfig.widths.includes(val)) {

                appConfig.widths.push(val); appConfig.widths.sort((a,b)=>a-b);

                saveAppData(); renderWidthList();

            }

        }


        function renderWidthList() {

            document.getElementById('widthList').innerHTML = appConfig.widths.map(w => `

                <div class="bg-black text-white px-3 py-1 rounded-xl font-bold flex items-center gap-2 text-xs">

                    ${w}mm <button onclick="appConfig.widths=appConfig.widths.filter(x=>x!=${w});saveAppData();renderWidthList();" class="text-red-500 font-black">&times;</button>

                </div>`).join('');

        }


        function openSettings() {

            document.getElementById('freq_nat_rell').value = appConfig.freq.natural.rell;

            document.getElementById('freq_nat_full').value = appConfig.freq.natural.full;

            document.getElementById('freq_sin_rell').value = appConfig.freq.sintetico.rell;

            document.getElementById('freq_sin_full').value = appConfig.freq.sintetico.full;

            renderWidthList();

            document.getElementById('settingsModal').style.display = 'block';

        }


        function exportData() {

            const data = { config: appConfig, bikes: bikes };

            const blob = new Blob([JSON.stringify(data)], {type: 'application/json'});

            const a = document.createElement('a');

            a.href = URL.createObjectURL(blob);

            a.download = `backup_bicis.json`;

            a.click();

        }


        function importData(e) {

            const reader = new FileReader();

            reader.onload = (e) => {

                const data = JSON.parse(e.target.result);

                if(data.config && data.bikes) {

                    appConfig = data.config; bikes = data.bikes;

                    saveAppData(); location.reload();

                }

            };

            reader.readAsText(e.target.files[0]);

        }


        function closeModal(id) { document.getElementById(id).style.display = 'none'; }

        function showOk() {

            const t = document.getElementById('toast');

            t.classList.remove('hidden');

            setTimeout(() => t.classList.add('hidden'), 3000);

        }


        function renderBikes() {

            const container = document.getElementById('bikeList');

            container.innerHTML = "";

            const today = new Date(); today.setHours(0,0,0,0);


            bikes.forEach((bike, index) => {

                const conf = bike.liquid === 'natural' ? appConfig.freq.natural : appConfig.freq.sintetico;

                const nextRell = new Date(bike.lastRefill); nextRell.setMonth(nextRell.getMonth() + conf.rell);

                const nextFull = new Date(bike.lastChange); nextFull.setMonth(nextFull.getMonth() + conf.full);

                const dRell = Math.ceil((nextRell - today) / 8.64e7);

                const dFull = Math.ceil((nextFull - today) / 8.64e7);

                const mlFull = calculateML(bike.category, bike.size, bike.width);

                const mlRell = Math.round(mlFull / 2);


                container.innerHTML += `

                    <div class="rounded-[2.5rem] shadow-xl p-6 bg-white ${dRell <= 0 || dFull <= 0 ? 'status-red' : 'status-green'} flex flex-col">

                        <div class="flex justify-between items-start mb-4 border-b pb-2">

                            <div class="w-full">

                                <h3 class="text-3xl font-black uppercase italic text-slate-900 leading-tight">${bike.name}</h3>

                                <p class="text-[10px] font-black text-blue-700 uppercase">${bike.category} | ${bike.size} x ${bike.width}</p>

                            </div>

                            <div class="flex gap-4">

                                <button onclick="openBikeModal(${index})" class="text-3xl text-slate-300 hover:text-slate-900"><i class="fas fa-edit"></i></button>

                                <button onclick="if(confirm('¿Borrar?')){bikes.splice(${index},1);saveAppData();}" class="text-3xl text-red-100 hover:text-red-600"><i class="fas fa-trash"></i></button>

                            </div>

                        </div>

                        <div class="space-y-4">

                            <div class="bg-blue-50 p-4 rounded-3xl border border-blue-100 flex justify-between items-center leading-none">

                                <div><p class="text-[9px] font-black uppercase text-blue-800 mb-1">Relleno</p><p class="text-ml-gigante">${mlRell}<span class="text-xs">ml</span></p></div>

                                <div class="text-right">

                                    <p class="text-[9px] font-black text-slate-400 uppercase">Faltan</p><p class="text-dias-gigante ${dRell <= 0 ? 'text-red-600 animate-pulse' : ''}">${dRell}d</p>

                                    <button onclick="quickUpdate(${index}, 'relleno')" class="mt-2 bg-blue-600 text-white px-6 py-2 rounded-xl font-black uppercase text-[10px]">Hecho</button>

                                </div>

                            </div>

                            <div class="bg-emerald-50 p-4 rounded-3xl border border-emerald-100 flex justify-between items-center leading-none">

                                <div><p class="text-[9px] font-black uppercase text-emerald-800 mb-1">Cambio</p><p class="text-ml-gigante">${mlFull}<span class="text-xs">ml</span></p></div>

                                <div class="text-right">

                                    <p class="text-[9px] font-black text-slate-400 uppercase">Faltan</p><p class="text-dias-gigante ${dFull <= 0 ? 'text-red-600 animate-pulse' : ''}">${dFull}d</p>

                                    <button onclick="quickUpdate(${index}, 'cambio')" class="mt-2 bg-emerald-600 text-white px-6 py-2 rounded-xl font-black uppercase text-[10px]">Hecho</button>

                                </div>

                            </div>

                        </div>

                    </div>`;

            });

        }

        renderBikes();
